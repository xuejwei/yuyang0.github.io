有用的代码片段

  1. count lines of code in a directory 
    find . -name '*.php' | xargs wc -l
# works on names with space
( find ./ -name '*.[h|c|cc|cpp|php]' -print0 | xargs -0 cat ) | wc -l
  2. 删除链表节点 
    void remove_if(node ** head, remove_fn rm)
{
    for (node** curr = head; *curr; )
    {
        node * entry = *curr;
        if (rm(entry))
        {
            *curr = entry->next;
            free(entry);
        }
        else
            curr = &entry->next;
    }
}
